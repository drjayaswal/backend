This is The NGINX Branch

apt update - it updates the machine to install the nginx, mostly Linux via ubuntu
apt-get install nginx - it downloads and installs the nginx onto the machine
cd etc/nginx && vim nginx.conf - navigate to the downloaded nginx's directory and updates the nginx code
nginx -s reload - reloads the nginx server, may requires 'sudo' command for permissions

nginx.conf:
    events {} - deals with how the server physically handles the network traffic at a low level
        worker_connections - handle 4096 connections per worker
        multi_accept - grab all waiting connections immediately
        use epoll - use the most efficient method for Linux
        example:
            events {
                worker_connections 4096;
                
                multi_accept on;
                
                use epoll;
            }
    http {} - the container for all web-related traffic, managing how data is packaged, compressed, and routed to specific sites
        include mime.types - maps file extensions to correct content types so browsers can render them
        default_type - sets the fallback for unknown files (usually a download action)
        sendfile - uses kernel-level memory management to send files faster without copying buffers
        keepalive_timeout - keeps connections open to allow multiple downloads without a new handshake
        gzip - compresses text-based files on the fly to save bandwidth and load faster
        access_log - records every visitor's request (IP, URL, status) to a specific file for debugging
        upstream - creates a named cluster of servers that Nginx can distribute traffic across
        include - imports modular site-specific files from other directories to keep the main file clean
        example:
            http {
                include       mime.types;

                default_type  application/octet-stream;

                sendfile      on;

                keepalive_timeout  65;

                gzip          on;

                access_log    /var/log/nginx/access.log;

                upstream my_app_cluster {
                    server 10.0.0.1:8080;
                    server 10.0.0.2:8080;
                }

                include /etc/nginx/sites-enabled/*;
            }
    server {} - handle requests for specific domains or IP addresses and sets global rules for all web traffic in the http context
        listen - listens on given port
        server_name - name specificed directive for distinguish between multi used ports
        root - shows the location of the website code/files
        return - shows the special NGINX code while the closing the connections
        example:
            server {
                listen 80;
                
                server_name abc.com www.abc.com;
                
                root /var/www/html;
                
                return 301 https://$server_name$request_uri;
            }

            docker pull <username>/<image-name>:latest                               
            
            # Start the application container in background mode, mapping ports for external access
            docker run -d -p <desired-machine-port>:<assigned-container-port> app:latest                         

        # Configure nginx to proxy requests to your Docker container
        $/etc/nginx/sites-enabled/
            # Edit the default nginx site configuration file
            sudo vim default                                              
                -- or --
            # Alternatively, append nginx configuration directly to the file
            echo -e "server {                                             
                        # Define how nginx handles incoming requests
                        location / {                                      
                            # Forward all root path requests to Docker container
                            proxy_pass http://127.0.0.1:<assigned-machine-port>;             
                            
                            # Forward API requests to backend Docker container
                            proxy_pass http://127.0.0.1:<assigned-machine-port>/api;             
                            
                            # Use HTTP/1.1 protocol for better performance with upstream servers
                            proxy_http_version 1.1;                       
                            
                            # Enable WebSocket support by passing upgrade headers
                            proxy_set_header Upgrade $http_upgrade;       
                            
                            # Maintain WebSocket connection state
                            proxy_set_header Connection 'upgrade';        
                            
                            # Preserve original host information for the backend
                            proxy_set_header Host $host;                  
                            # Disable caching for WebSocket connections
                            proxy_cache_bypass $http_upgrade;             
                        }
                    }" >> default                                         

        # Test the complete setup to ensure everything works correctly
        $ubuntu
            # Verify that your application is accessible through nginx proxy
            curl -L http://127.0.0.1:<assigned-machine-port>
